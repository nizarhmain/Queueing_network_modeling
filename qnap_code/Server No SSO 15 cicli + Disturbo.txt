    986 	    NRTEST:=N_Router;
    987 	    NVTEST:=N_Vie;
    988 	    NSOFTW:=N_soft;
    989 	
    990 	    LINE_CAP:=GET(REAL);
    991 	    REC_CAP:= GET(REAL);
    992 	    CAP_BUF:= GET(REAL);
    993 	    NewLine;
    994 	    T_MAX:=GET(REAL);
    995 	    TEMPO := GET(REAL);
    996 	    PERIODO:=GET(REAL);
    997 	    NewLine;
    998  
    999 & GENERAZIONE ROUTERS
   1000  
   1001 	  IF (N_Router > 0) THEN
   1002 	       FOR I:=1 STEP 1 UNTIL N_Router DO BEGIN
   1003 	       idr :=GET(INTEGER);
   1004 	       ROUTER#(idr):=NEW(ROUTER,idr,GET(INTEGER));
   1005 	       WITH ROUTER#(idr) DO BEGIN
   1006 	       ROUTER#(idr).RATE := GET(REAL);
   1007 	       ROUTER#(idr).P_ERR := GET(REAL);
   1008 	       END;
   1009 	 NewLine;
   1010  END;
   1011  
   1012 & GENERAZIONE VIE E CONNESSIONI A ROUTER DI USCITA
   1013  
   1014  IF (N_Vie > 0) THEN
   1015    FOR I:=1 STEP 1 UNTIL N_Vie DO BEGIN
   1016    idv :=GET(INTEGER);
   1017    VIA#(idv):=NEW(VIA,idv,GET(REAL));
   1018       WITH VIA#(idv) DO BEGIN
   1019       idr:=GET(INTEGER);
   1020       NEXT_R:=ROUTER#(idr).CPU;
   1021       END;
   1022  NewLine;
   1023  END;
   1024  
   1025 & ISTRADAMENTI ROUTER->ROUTER
   1026  
   1027     FOR I:=1 STEP 1 UNTIL N_Router DO BEGIN
   1028     idr:=GET(INTEGER);
   1029 	WITH ROUTER#(idr) DO BEGIN
   1030 	     FOR J:=1 STEP 1 UNTIL N DO BEGIN
   1031 	 	K:=GET(INTEGER);
   1032 		INSTR(J):=VIA#(K).CANALE;
   1033 		PROB(J):=GET(REAL);
   1034 		END;
   1035                 END;
   1036 	NewLine;
   1037     END;
   1038  
   1039 & GENERAZIONE HOSTS E LINE_HR
   1040  
   1041    IF (N_Hosts > 0) THEN
   1042    FOR I:=1 STEP 1 UNTIL N_Hosts DO BEGIN
   1043         idh:=GET(INTEGER);
   1044         HOSTS#(idh):=NEW(HOST,idh);
   1045         WITH HOSTS#(idh) DO BEGIN
   1046  
   1047 	   HOSTS#(idh).T_EMI_I := GET(REAL);
   1048 	   HOSTS#(idh).T_EMI_A := GET(REAL);
   1049 	   HOSTS#(idh).T_REC_I := GET(REAL);
   1050 	   HOSTS#(idh).T_REC_A := GET(REAL);
   1051 	   HOSTS#(idh).TIME_OUT := GET(REAL);
   1052 	   HOSTS#(idh).WIN_SIZE := GET(INTEGER);
   1053 	   HOSTS#(idh).LI := GET(INTEGER);
   1054 	   HOSTS#(idh).SSO := GET(INTEGER);
   1055 	   HOSTS#(idh).LA := GET(INTEGER);
   1056 	   END;
   1057  
   1058 	LINE_HR#(idh):=NEW(LINE_HR,GET(REAL));
   1059  
   1060 & CONNESSIONI ROUTER-HOST
   1061  
   1062     idr:=GET(INTEGER);
   1063     HOSTS#(idh).ID_R := idr;
   1064     ROUTER#(idr).S(idh):=HOSTS#(idh).ACCESS_R;
   1065     NewLine;
   1066     END;
   1067  
   1068 & CONNESSIONI HOST-ROUTER
   1069  
   1070       IF (N_Hosts > 0) THEN
   1071       FOR I:=1 STEP 1 UNTIL N_Hosts DO BEGIN
   1072 	   HOSTS#(I).NETWORK:=LINE_HR#(I).R;
   1073 	
   1074          LINE_HR#(I).ROUT:=ROUTER#(HOSTS#(I).ID_R).CPU;
   1075 	 END;
   1076  
   1077  
   1078 &--------------------------------------------------
   1079 & COSTRUZIONE TERMINALE E CONNESSIONI TERMINAL-HOST
   1080 &--------------------------------------------------
   1081  
   1082 IF (N_Hosts > 0) THEN
   1083 FOR I:=1 STEP 1 UNTIL N_Hosts DO
   1084 BEGIN
   1085   	idter := GET(INTEGER);
   1086 	TERM#(idter):=NEW(TERMINAL,idter);
   1087      	WITH TERM#(idter) DO BEGIN
   1088            TERM#(idter).ACTIVES := GET(BOOLEAN);
   1089            TERM#(idter).NAME := GET(STRING);
   1090 	   PRINT ("GEN",TERM#(idter).ACTIVES, TERM#(idter).NAME);
   1091 	   TERM#(idter).REQLENGHT := GET(INTEGER);
   1092            TERM#(idter).PROCESS := GET(REAL);
   1093 	   TERM#(idter).PROC_D := GET(REAL);
   1094      	END;
   1095         LINE_TH#(idter):=NEW(LINE_TH,GET(REAL));
   1096 	
   1097 & CONNESSIONI HOST-TERMINAL
   1098  
   1099 	 idhost:=GET(INTEGER);
   1100 	 TERM#(idter).ID_H := idhost;
   1101          HOSTS#(idhost).HOTE:=LINE_TH#(idter).HQ;
   1102 	 LINE_TH#(idter).TIN:=TERM#(idter).ACCESS_H;
   1103          NewLine;
   1104 	END;
   1105  
   1106  
   1107 & CONNESSIONI TERMINAL-HOST
   1108  
   1109         IF (N_Hosts > 0) THEN
   1110         FOR I:=1 STEP 1 UNTIL N_Hosts DO BEGIN
   1111 	   TERM#(I).PORTA:=LINE_TH#(I).QH;
   1112 	   LINE_TH#(I).TOUT:=HOSTS#(TERM#(I).ID_H).USER;
   1113 END;
   1114  
   1115 &--------------------------------------------------
   1116 & COSTRUZIONE ARRY SOFTWARE
   1117 &--------------------------------------------------
   1118  IF (N_soft > 0) THEN
   1119   FOR I:=1 STEP 1 UNTIL (N_soft) DO BEGIN
   1120   	SW#(I):=NEW (SOFTWARE);
   1121   	SW#(I).IDS:=GET(INTEGER);
   1122   	SW#(I).EXIT:=GET(INTEGER);
   1123   	IF (SW#(I).IDS<>0) THEN BEGIN
   1124   		SW#(I).FA:= NEW (FILE);
   1125   		FILASSIGN(SW#(I).FA,GET(STRING));
   1126   		END
   1127   	ELSE
   1128 		IF(HOSTS#(SW#(I).EXIT).SSO=1)THEN BEGIN
   1129   		SW#(I).FA:= NEW (FILE);
   1130   		FILASSIGN(SW#(I).FA,GET(STRING));
   1131   		END
   1132 		ELSE
   1133   		SW#(I).FA:= SW#(GET(INTEGER)).FA;
   1134   	NewLine;
   1135   	END;
   1136  
   1137  
   1138 &-------------
   1139         END;
