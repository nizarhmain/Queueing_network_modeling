    162  PROCEDURE COPY_P(C,SOUR,DEST);
    163 	INTEGER SOUR,DEST;
    164 	REF CUSTOMER C;
    165 	REF HOST RH;
    166 	BEGIN
    167 		WITH CUSTOMER DO BEGIN
    168 		RH:= INCLUDING (QUEUE)::HOST;
    169 		C.DEST_ID:= DEST;
    170 		C.SOUR_ID:= SOUR;
    171 		TRANSIT(C,RH.TIMER);
    172 		END;
    173 	END;

    215  PROCEDURE PROT_EMI(TEI,TEA);
    216 	REAL TEI,TEA;
    217 	REF HOST RH;
    218 	BEGIN
    219 		WITH CUSTOMER DO BEGIN

    240  PROCEDURE PROT_REC(TRI,TRA,HOST_ID);
    241 	REAL TRI,TRA;
    242 	INTEGER HOST_ID;
    243 	REF HOST RH;
    244 	BEGIN
    245 		WITH CUSTOMER DO BEGIN
    246 		RH:= INCLUDING (QUEUE)::HOST;
    247 	IF TYP = "info" THEN
    248 		BEGIN
    249 			IF TYPE = "rich" THEN
    250 				BEGIN
    251 					TYP:="ack";
    252 					TYPE:="risp";
    253 					DEST_ID:= SOUR_ID;
    254 					SOUR_ID:= HOST_ID;
    255 					EXP(TRI);
    256 					COPY_RIC(NEW(CUSTOMER),SOUR_ID,DEST_ID);
    257 					TRANSIT(RH.PROTOCOL,EMISSION);
    258 				END
    259 			ELSE
    260 				BEGIN
    261 					TYP:="ack";
    262 					DEST_ID:= SOUR_ID;
    263 					SOUR_ID:=HOST_ID;
    264 					EXP(TRI);
    265 					COPY_TER(NEW(CUSTOMER),SOUR_ID,DEST_ID);
    266 					TRANSIT(RH.PROTOCOL,EMISSION);
    267 				END;
    268 		END
    269 	ELSE
    270 		BEGIN
    271 			EXP(TRA);
    272 				IF (HOST.TIMER.FIRST<> NIL)THEN
    273 					BEGIN
    274 					TRANSIT(RH.TIMER.FIRST,OUT);
    275 					V(RH.WINDOW);
    276 					END;
    277 				TRANSIT(OUT);
    278 			END;
    279 		END;
    280 	END;
